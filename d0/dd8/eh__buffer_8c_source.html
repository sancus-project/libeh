00052 ssize_t eh_buffer_read(struct eh_buffer *self, int fd)
00053 {
00054         assert(self != NULL);
00055         assert(fd >= 0);
00056 
00057         if (eh_buffer_free(self) == 0)
00058                 return 0; /* full */
00059 
00060         /* maintainance */
00061         if (self->base > 0) {
00062                 if (self->len == 0)
00063                         eh_buffer_reset(self);
00064                 else if (self->base + self->len == self->size)
00065                         __eh_buffer_rebase(self);
00066         }
00067 
00068         return read(fd, eh_buffer_next(self), eh_buffer_freetail(self));
00069 }

00071 ssize_t eh_buffer_write(struct eh_buffer *self, int fd)
00072 {
00073         ssize_t l;
00074         assert(self != NULL);
00075         assert(fd >= 0);
00076 
00077         if (eh_buffer_len(self) == 0) {
00078                 eh_buffer_reset(self);
00079                 return 0; /* empty */
00080         }
00081 
00082         l = write(fd, eh_buffer_data(self), eh_buffer_len(self));
00083         if (l == (ssize_t)eh_buffer_len(self))
00084                 eh_buffer_reset(self);

00086                 self->len -= l;

00088

00091                         __eh_buffer_rebase(self);
00092         }
