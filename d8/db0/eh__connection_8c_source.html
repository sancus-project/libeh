00045 {
00046         struct eh_connection *self = w->data;

00052 
00053         if (revents & EV_READ) {
00054                 struct eh_buffer *buf = &self->read_buffer;
00055                 ssize_t l;
00056 
00057                 if (eh_buffer_free(buf) == 0) {
00058                         bool close = true;
00059                         if (cb->on_error)
00060                                 close = cb->on_error(self, EH_CONNECTION_READ_FULL);
00061                         if (close)

00066                 l = eh_buffer_read(buf, w->fd);
00067 
00068                 if (l == 0) { /* EOF */
00069                         goto terminate;
00070                 } else if (l > 0) { /* has new data, pass over */
00071                         ssize_t rc = eh_buffer_len(buf);

00073                                 rc = cb->on_read(self, eh_buffer_data(buf), rc);
